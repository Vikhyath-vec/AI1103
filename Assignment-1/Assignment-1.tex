\section*{Question}
A box contains 5 red marbles, 8 white marbles, and 4 green marbles. One marble is taken out of the box at random. What is the probability that the marble taken out will be
\begin{enumerate}
    \item red?
    \item white?
    \item not green?
\end{enumerate}

\section*{Solution}
total number of marbles = 5 + 8 + 4 = 17 marbles
\begin{enumerate}
    

\item Probability that the marble taken out will be red = P(red)
\begin{align}
    P(red) = \frac{\mbox{number of red marbles}}{\mbox{total number of marbles}}\\
    P(red) = \frac{5}{17} = 0.294117647
\end{align}
\item Probability that the marble taken out will be white = P(white)
\begin{align}
    P(white) = \frac{\mbox{number of white marbles}}{\mbox{total number of marbles}}\\
    P(white) = \frac{8}{17} = 0.4705882353
\end{align}
\item Probability that the marble taken out will not be green = P(not green)
\\P(not green) + P(green) = 1 (since complementary events are mutually exclusive)
\\$\implies$ P(not green) = 1 - P(green)
\begin{align}
    P(green) = \frac{\mbox{number of green marbles}}{\mbox{total number of marbles}}\\
    P(green) = \frac{4}{17} = 0.235294117\\
    \implies P(\mbox{not green}) = 1 - 0.235294117 = 0.764705883
\end{align}

\end{enumerate}
